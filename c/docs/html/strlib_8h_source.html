1 /*
2  * File: strlib.h
3  * --------------
4  * This interface defines a general library for dynamically allocated
5  * strings. The major differences between traditional C strings and
6  * those defined using this interface are:
7  *
8  *<ol>
9  * <li>The <code>strlib.h</code> interface takes care of memory
10  * allocation, ensuring that there is sufficient space to hold
11  * the result of each string operation.
12  * <li>Clients of the <code>strlib.h</code> interface are expected
13  * to treat all strings as immutable and refrain from writing
14  * into the character array.
15  *</ol>
16  */

36 /*
37  * Cautionary note:
38  * ----------------
39  * Although this interface provides an extremely convenient abstraction
40  * for working with strings, it is not appropriate for all applications.
41  * In this interface, the functions that return string values (such as
42  * concat and substring) do so by allocating new memory. Over time, a
43  * program that uses this package will consume increasing amounts of
44  * memory and eventually exhaust the available supply. If you are
45  * writing a program that runs for a short time and stops, the fact
46  * that the package consumes memory is not a problem. If, however,
47  * you are writing an application that must run for an extended period
48  * of time, using this package requires that you make some provision
49  * for freeing any allocated storage.
50  */
