1 /*
2  * File: exception.h
3  * -----------------
4  * This interface exports a portable exception-handling mechanism for C.
5  *
6  * <p>The <code>exception.h</code> interface makes it possible for
7  * clients to specify a handler for an exceptional conditions in a
8  * syntactically readable way. As a client, your first step
9  * is to declare an exception condition name by declaring
10  * a variable of type exception, as in
11  *
12  *<pre>
13  * exception MyException;
14  *</pre>
15  *
16  * <p>Normal visibility rules apply, so that you should declare the
17  * exception variable at the appropriate level. For example, if an
18  * exception is local to an implementation, it should be declared
19  * statically within that module. If an exception condition is
20  * shared by many modules, the exception variable should be declared
21  * in an interface and exported to all clients that need it. This
22  * package defines and exports the exception <code>ErrorException</code>,
23  * which is likely to be sufficient for many clients.
24  *
25  * <p>The basic functionality of exceptions is that one piece of code
26  * can "throw" an exception so that it can then be "caught" by
27  * special code in a dynamically enclosing section of the program.
28  * Exceptions are triggered by calling the pseudo-function
29  * <code>throw</code> with the exception name, as in
30  *
31  *<pre>
32  * throw(MyException);
33  *</pre>
34  *
35  * <p>Exceptions are handled using the <code>try</code> statement
36  * (actually implemented using macros), which has the form:
37  *
38  *<pre>
39  * try {
40  * . . . statements in the body of the block . . .
41  * } catch (exception1) {
42  * . . . statements to handle exception 1 . . .
43  * } catch (exception2) {
44  * . . . statements to handle exception 2 . . .
45  * } catch (ANY) {
46  * . . . statements to handle any exception . . .
47  * } finally {
48  * . . . statements to be executed before exit in all cases . . .
49  * } endtry
50  *</pre>
51  *
52  * <p>Any number of <code>catch</code> clauses may appear. The
53  * <code>ANY</code> and <code>finally</code> clauses are optional.
54  *
55  * <p>When the program encounters the <code>try</code> statement, the
56  * statements in the body are executed. If no exception conditions
57  * are thrown during that execution, either in this block or by a
58  * function call nested inside this block, control passes to the end
59  * of the <code>try</code> statement when the last statement in the
60  * block is executed. If an exception is thrown during the dynamic
61  * execution of the block, control immediately passes to the statements
62  * in the appropriate <code>catch</code> clause. Only the statements
63  * in that clause are executed; no <code>break</code> statement is
64  * required to exit the block.
65  *
66  * <p>The <code>try</code> statement guarantees that the statements in
67  * the <code>finally</code> clause, if present, will always be executed,
68  * even if an exception is thrown that is caught at a higher level.
69  *
70  * <p>If no handler for the exception appears anywhere in the
71  * control history, the program exits with an error.
72  *
73  * <p>Examples of use:
74  *
75  * <p>1. Catching errors
76  *
77  * <p>The following code fragment traps calls to <code>error</code>, so
78  * that the program does not quit but instead returns to the top-level
79  * read-and-execute loop.
80  *
81  *<pre>
82  * while (true) {
83  * try {
84  * printf("> ");
85  * cmd = readCommand();
86  * executeCommand(cmd);
87  * } catch (ErrorException) {
88  * printf("error: %s\n", (string) getExceptionValue());
89  * -- additional handling code, if any --
90  * } endtry
91  * }
92  *</pre>
93  *
94  * <p>If either <code>readCommand</code> or <code>executeCommand</code>
95  * calls <code>error</code>, control will be passed back to the main
96  * loop, after executing any additional handler code. The error
97  * message is passed as the exception value and can be printed as
98  * shown in the example.
99  *
100  * <p>2. Handling control-C (Unix systems)
101  *
102  * <p>The following code extends the example above so that typing
103  * <code>^C</code> also returns to top-level:
104  *
105  *<pre>
106  * #include <signal.h>
107  * &nbsp;
108  * static exception ControlCException;
109  * static int errorCount = 0;
110  * static int controlCHandler();
111  * &nbsp;
112  * main() {
113  * string cmd;
114  * &nbsp;
115  * signal(SIGINT, controlCHandler);
116  * while (true) {
117  * try {
118  * printf("> ");
119  * cmd = readCommand();
120  * executeCommand(cmd);
121  * } catch (ControlCException) {
122  * printf("^C\n");
123  * signal(SIGINT, controlCHandler);
124  * } catch (ErrorException) {
125  * errorCount++;
126  * } endtry
127  * }
128  * }
129  * &nbsp;
130  * static int controlCHandler() {
131  * throw(ControlCException);
132  * }
133  *</pre>
134  */
