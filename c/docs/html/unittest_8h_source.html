35 /*
36  * Function: trace
37  * Usage: trace(exp);
38  * ------------------
39  * Tests to make sure that a procedure does not generate any calls
40  * to <code>error</code>.
41  */
42 
43 #define trace(exp) { \
44  try { \
45  reportMessage("%s;", #exp); \
46  exp; \
47  } catch (ErrorException) { \
48  reportError("%s throws \"%s\"", #exp, getExceptionValue()); \
49  } endtry \
50 }

52 /*
53  * Function: test
54  * Usage: test(exp, value);
55  * ------------------------
56  * Tests to make sure that a function computes a specified value. The
57  * first argument is an expression (typically a function call in the
58  * module under test), and the second argument is the expected value,
59  * which must be of one of the following types:
60  *
61  * <p><ul>
62  * <li><code>bool</code>
63  * <li><code>char</code>
64  * <li><code>double</code>
65  * <li><code>int</code>
66  * <li><code>string</code>
67  * </ul>
68  */
69 
70 #define test(exp, value) { \
71  try { \
72  uTTest(#exp, #value, exp, value); \
73  } catch (ErrorException) { \
74  reportError("%s throws \"%s\"", #exp, getExceptionValue()); \
75  } endtry \
76 }

78 /*
79  * Function: testError
80  * Usage: testError(exp);
81  * ----------------------
82  * Tests to make sure that a method call generates an error.
83  */
84 
85 #define testError(exp) { \
86  int indent = getReportIndentation(); \
87  try { \
88  exp; \
89  setReportIndentation(indent); \
90  reportError("%s should generate an error", #exp); \
91  } catch (ErrorException) { \
92  setReportIndentation(indent); \
93  reportMessage("%s throws \"%s\"", #exp, getExceptionValue()); \
94  } catch (ANY) { \
95  setReportIndentation(indent); \
96  reportMessage("%s throws \"%s\"", #exp, getExceptionName()); \
97  } endtry \
98 }
